  sliders-horizontal sort-ascending sort-descending speaker-high speaker-low
  speaker-x spinner star stop terminal-window trash trash-simple upload user
  user-check user-circle user-minus user-plus users video-camera warning
  warning-circle warning-diamond warning-octagon wifi-high wifi-slash wrench x-circle
)

# Fetch with progress
TOTAL=${#NAMES[@]}
echo "Fetching $TOTAL SVGs..."
i=0
FAILED=()
for name in "${NAMES[@]}"; do
  i=$((i+1))
  if [ -f "${DEST}/${name}.svg" ]; then
    echo "[$i/$TOTAL] ${name}.svg already vendored — skipping"
    continue
  fi
  if curl -fsSL "${BASE}/${name}.svg" -o "${DEST}/${name}.svg"; then
    echo "[$i/$TOTAL] OK ${name}.svg"
  else
    echo "[$i/$TOTAL] FAIL ${name}.svg"
    FAILED+=("${name}")
    rm -f "${DEST}/${name}.svg"
  fi
done

if [ ${#FAILED[@]} -gt 0 ]; then
  echo "FAILED FETCHES:"
  printf '  %s\n' "${FAILED[@]}"
  echo "These names returned non-200 at the pinned SHA. Either the master list has typos, or the SHA is missing these icons. Halt and investigate before proceeding to Task 3."
  exit 1
fi
